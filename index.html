        // ==================== UTILITY FUNCTIONS ====================

        

        // Format number for display

        function formatNumber(num) {

            return parseFloat(num).toFixed(2);

        }


        // Convert radians to degrees

        function radToDeg(rad) {

            return rad * (180 / Math.PI);

        }


        // Clamp value between min and max

        function clamp(value, min, max) {

            return Math.min(Math.max(value, min), max);

        }


        // ==================== CUSTOM HOOKS (State Management) ====================

        

        class ViewerState {

            constructor() {

                this.camera = null;

                this.scene = null;

                this.renderer = null;

                this.model = null;

                this.controls = null;

                this.lights = {};

                this.autoRotate = true;

                this.rotationSpeed = 0.01;

                this.subscribers = [];

            }


            subscribe(callback) {

                this.subscribers.push(callback);

            }


            notify(data) {

                this.subscribers.forEach(callback => callback(data));

            }


            updateCameraInfo() {

                if (this.camera) {

                    const pos = this.camera.position;

                    const rot = this.camera.rotation;

                    const zoom = this.camera.zoom || 1;

                    

                    this.notify({

                        type: 'camera',

                        position: { x: pos.x, y: pos.y, z: pos.z },

                        rotation: { 

                            x: radToDeg(rot.x), 

                            y: radToDeg(rot.y), 

                            z: radToDeg(rot.z) 

                        },

                        zoom: zoom

                    });

                }

            }

        }

        class ControlPanel {

            constructor(state) {

                this.state = state;

                this.setupEventListeners();

            }


            setupEventListeners() {

                // Camera controls

                document.getElementById('posX').addEventListener('input', (e) => {

                    if (this.state.camera) {

                        this.state.camera.position.x = parseFloat(e.target.value);

                        this.state.updateCameraInfo();

                    }

                });


                document.getElementById('posY').addEventListener('input', (e) => {

                    if (this.state.camera) {

                        this.state.camera.position.y = parseFloat(e.target.value);

                        this.state.updateCameraInfo();

                    }

                });


                document.getElementById('posZ').addEventListener('input', (e) => {

                    if (this.state.camera) {

                        this.state.camera.position.z = parseFloat(e.target.value);

                        this.state.updateCameraInfo();

                    }

                });


                document.getElementById('resetCamera').addEventListener('click', () => {

                    this.resetCamera();

                });


                // Material controls

                document.getElementById('materialColor').addEventListener('input', (e) => {

                    this.updateMaterialColor(e.target.value);

                });


                document.getElementById('metalness').addEventListener('input', (e) => {

                    this.updateMaterialProperty('metalness', parseFloat(e.target.value));

                });


                document.getElementById('roughness').addEventListener('input', (e) => {

                    this.updateMaterialProperty('roughness', parseFloat(e.target.value));

                });


                document.getElementById('emissive').addEventListener('input', (e) => {

                    this.updateEmissive(parseFloat(e.target.value));

                });


                // Model controls

                document.getElementById('modelScale').addEventListener('input', (e) => {

                    this.updateModelScale(parseFloat(e.target.value));

                });


                document.getElementById('rotationSpeed').addEventListener('input', (e) => {

                    this.state.rotationSpeed = parseFloat(e.target.value);

                });


                document.getElementById('toggleRotation').addEventListener('click', () => {

                    this.toggleAutoRotation();

                });


                // Lighting controls

                document.getElementById('lightIntensity').addEventListener('input', (e) => {

                    this.updateLightIntensity(parseFloat(e.target.value));

                });


                document.getElementById('ambientLight').addEventListener('input', (e) => {

                    this.updateAmbientLight(parseFloat(e.target.value));

                });

            }


            resetCamera() {

                if (this.state.camera) {

                    this.state.camera.position.set(0, 0, 5);

                    this.state.camera.rotation.set(0, 0, 0);

                    this.state.updateCameraInfo();

                    

                    // Update sliders

                    document.getElementById('posX').value = 0;

                    document.getElementById('posY').value = 0;

                    document.getElementById('posZ').value = 5;

                }

            }


            updateMaterialColor(color) {

                if (this.state.model && this.state.model.material) {

                    this.state.model.material.color.setHex(color.replace('#', '0x'));

                }

            }


            updateMaterialProperty(property, value) {

                if (this.state.model && this.state.model.material) {

                    this.state.model.material[property] = value;

                }

            }


            updateEmissive(intensity) {

                if (this.state.model && this.state.model.material) {

                    const color = this.state.model.material.color.clone();

                    this.state.model.material.emissive = color.multiplyScalar(intensity);

                }

            }


            updateModelScale(scale) {

                if (this.state.model) {

                    this.state.model.scale.set(scale, scale, scale);

                }

            }


            toggleAutoRotation() {

                this.state.autoRotate = !this.state.autoRotate;

                const button = document.getElementById('toggleRotation');

                button.textContent = this.state.autoRotate ? 'Stop Auto Rotation' : 'Start Auto Rotation';

            }


            updateLightIntensity(intensity) {

                if (this.state.lights.directional) {

                    this.state.lights.directional.intensity = intensity;

                }

            }


            updateAmbientLight(intensity) {

                if (this.state.lights.ambient) {

                    this.state.lights.ambient.intensity = intensity;

                }

            }

        }


        // ==================== FEATURE COMPONENTS ====================

        

        class ThreeJSViewer {

            constructor(container, state) {

                this.container = container;

                this.state = state;

                this.mouseControls = {

                    isDragging: false,

                    previousMousePosition: { x: 0, y: 0 }

                };

                this.init();

            }


            init() {

                this.setupScene();

                this.setupCamera();

                this.setupRenderer();

                this.setupLights();

                this.setupModel();

                this.setupControls();

                this.startRenderLoop();

                

                // Hide loading indicator

                document.getElementById('loading').style.display = 'none';

            }


            setupScene() {

                this.state.scene = new THREE.Scene();

                this.state.scene.background = new THREE.Color(0x222222);

                

                // Add fog for depth perception

                this.state.scene.fog = new THREE.Fog(0x222222, 1, 100);

            }


            setupCamera() {

                this.state.camera = new THREE.PerspectiveCamera(

                    75,

                    this.container.clientWidth / this.container.clientHeight,

                    0.1,

                    1000

                );

                this.state.camera.position.set(0, 0, 5);

            }


            setupRenderer() {

                this.state.renderer = new THREE.WebGLRenderer({ 

                    antialias: true,

                    alpha: true 

                });

                this.state.renderer.setSize(this.container.clientWidth, this.container.clientHeight);

                this.state.renderer.setPixelRatio(window.devicePixelRatio);

                this.state.renderer.shadowMap.enabled = true;

                this.state.renderer.shadowMap.type = THREE.PCFSoftShadowMap;

                this.state.renderer.toneMapping = THREE.ACESFilmicToneMapping;

                this.state.renderer.toneMappingExposure = 1;

                this.container.appendChild(this.state.renderer.domElement);


                // Handle window resize

                window.addEventListener('resize', () => this.handleResize());

            }


            setupLights() {

                // Ambient light

                this.state.lights.ambient = new THREE.AmbientLight(0x404040, 0.4);

                this.state.scene.add(this.state.lights.ambient);


                // Directional light

                this.state.lights.directional = new THREE.DirectionalLight(0xffffff, 1);

                this.state.lights.directional.position.set(5, 5, 5);

                this.state.lights.directional.castShadow = true;

                this.state.lights.directional.shadow.mapSize.width = 2048;

                this.state.lights.directional.shadow.mapSize.height = 2048;

                this.state.scene.add(this.state.lights.directional);


                // Point light for additional illumination

                const pointLight = new THREE.PointLight(0xffffff, 0.5, 100);

                pointLight.position.set(-5, 5, 5);

                this.state.scene.add(pointLight);

            }


            setupModel() {

                // Create a more interesting default model (torus knot)

                const geometry = new THREE.TorusKnotGeometry(1, 0.3, 100, 16);

                const material = new THREE.MeshStandardMaterial({

                    color: 0x4CAF50,

                    metalness: 0.1,

                    roughness: 0.8,

                    envMapIntensity: 1

                });


                this.state.model = new THREE.Mesh(geometry, material);

                this.state.model.castShadow = true;

                this.state.model.receiveShadow = true;

                this.state.scene.add(this.state.model);


                // Add a ground plane

                const groundGeometry = new THREE.PlaneGeometry(20, 20);

                const groundMaterial = new THREE.MeshStandardMaterial({ 

                    color: 0x999999, 

                    transparent: true, 

                    opacity: 0.3 

                });

                const ground = new THREE.Mesh(groundGeometry, groundMaterial);

                ground.rotation.x = -Math.PI / 2;

                ground.position.y = -2;

                ground.receiveShadow = true;

                this.state.scene.add(ground);

            }


            setupControls() {

                const canvas = this.state.renderer.domElement;


                // Mouse controls

                canvas.addEventListener('mousedown', (e) => this.onMouseDown(e));

                canvas.addEventListener('mousemove', (e) => this.onMouseMove(e));

                canvas.addEventListener('mouseup', () => this.onMouseUp());

                canvas.addEventListener('wheel', (e) => this.onWheel(e));


                // Touch controls for mobile

                canvas.addEventListener('touchstart', (e) => this.onTouchStart(e));

                canvas.addEventListener('touchmove', (e) => this.onTouchMove(e));

                canvas.addEventListener('touchend', () => this.onTouchEnd());

            }


            onMouseDown(event) {

                this.mouseControls.isDragging = true;

                this.mouseControls.previousMousePosition = {

                    x: event.clientX,

                    y: event.clientY

                };

            }


            onMouseMove(event) {

                if (!this.mouseControls.isDragging) return;


                const deltaMove = {

                    x: event.clientX - this.mouseControls.previousMousePosition.x,

                    y: event.clientY - this.mouseControls.previousMousePosition.y

                };


                // Rotate the model based on mouse movement

                if (this.state.model) {

                    this.state.model.rotation.y += deltaMove.x * 0.01;

                    this.state.model.rotation.x += deltaMove.y * 0.01;

                }


                this.mouseControls.previousMousePosition = {

                    x: event.clientX,

                    y: event.clientY

                };


                this.state.updateCameraInfo();

            }


            onMouseUp() {

                this.mouseControls.isDragging = false;

            }


            onWheel(event) {

                event.preventDefault();

                const delta = event.deltaY > 0 ? 1.1 : 0.9;

                this.state.camera.position.multiplyScalar(delta);

                this.state.updateCameraInfo();

            }


            onTouchStart(event) {

                if (event.touches.length === 1) {

                    this.onMouseDown({

                        clientX: event.touches[0].clientX,

                        clientY: event.touches[0].clientY

                    });

                }

            }


            onTouchMove(event) {

                if (event.touches.length === 1) {

                    this.onMouseMove({

                        clientX: event.touches[0].clientX,

                        clientY: event.touches[0].clientY

                    });

                }

            }


            onTouchEnd() {

                this.onMouseUp();

            }


            handleResize() {

                const width = this.container.clientWidth;

                const height = this.container.clientHeight;


                this.state.camera.aspect = width / height;

                this.state.camera.updateProjectionMatrix();


                this.state.renderer.setSize(width, height);

            }


            startRenderLoop() {

                const animate = () => {

                    requestAnimationFrame(animate);


                    // Auto rotation

                    if (this.state.autoRotate && this.state.model) {

                        this.state.model.rotation.y += this.state.rotationSpeed;

                    }


                    this.state.renderer.render(this.state.scene, this.state.camera);

                    this.state.updateCameraInfo();

                };


                animate();

            }

        }
